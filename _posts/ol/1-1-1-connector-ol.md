wax.tilejson('http://c.tiles.mapbox.com/v3/examples.map-vyofok3q.jsonp',
function(tilejson) {
    var map = new OpenLayers.Map({
        div: 'map-div',
        controls: [
            new OpenLayers.Control.Navigation(),
            new OpenLayers.Control.Attribution()
        ],
        layers: [
            wax.ol.connector(tilejson)
        ]
    });
    map.zoomTo(3);
});
</pre>
